async function guardarDatos() { const nombre = document.getElementById("nombre").value; const correo = document.getElementById("correo").value; if (!nombre || !correo) { alert("¡Oops! Completa ambos campos."); return; } try { const response = await fetch("https://script.google.com/macros/s/AKfycbzxqZof2dO5MywKalHEWSU9B59K49P0IEz3BepWkioORnGBimPC07iDAR8_yb2ewvPnvQ/exec", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ nombre, correo }), }); const result = await response.json(); if (result.success) { document.getElementById("mensaje").innerHTML = `¡Gracias, ${nombre}!
Datos guardados en Google Sheets.`; } else { throw new Error(result.error || "Error desconocido"); } } catch (error) { document.getElementById("mensaje").innerHTML = `Hubo un error: ${error.message}
Por favor, recarga la página e intenta nuevamente.`; console.error("Error:", error); } }
